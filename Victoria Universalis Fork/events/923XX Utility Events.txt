	is_triggered_only = yes

		name = EVTOPTA92305
	}
	
	# Ask to unoccupy country

		name = EVTOPTB92305
		
		any_owned = {
			limit = {
				NOT = { controlled_by = THIS }
				controller = { ai = no }
			}
			
			controller = {
				country_event = {
					id = 92306
					days = 0
				}
			}
		}
	}
	
	# Ask to occupy enemy capital
	option = {
		name = EVTOPTC92305
		
		any_country = {
			limit = {
				ai = no
				war_with = THIS 
			}
			
			country_event = {
				id = 92307
				days = 0
			}
		}
	}
}
