# Base Province Selector Event
province_event = {
	id = 92400
	title = EVTNAME92400
	desc = EVTDESC92400
	
	is_triggered_only = yes
	
	immediate = {
		owner = {
			any_owned = {
				limit = {
					has_building = province_selector
				}
				
				province_selector = -1
			}
		}
	}
	
	# Cancel
	option = {
		name = EVTOPTA92400
	}
	
	# Add/Remove Migration Attraction
	option = {
		name = EVTOPTB92400
		
		add_core = JAN
		
		# Remove modifier
		owner = {
			random_owned = {
				limit = {
					is_core = JAN
					
					has_province_modifier = province_attract_immigrants
				}
				
				remove_province_modifier = province_attract_immigrants
			}
		}
		
		# Add modifier
		owner = {
			random_owned = {
				limit = {
					is_core = JAN
					is_colonial = no
					
					NOT = { has_province_modifier = province_attract_immigrants }
				}
				
				add_province_modifier = {
					name = province_attract_immigrants
					duration = -1
				}
			}
		}
		
		remove_core = JAN
	}
	
	# Make state a colony
	option = {
		name = EVTOPTC92400
		
		add_core = JAN
		
		owner = {
			random_owned = {
				limit = {
					is_core = JAN
					is_colonial = no
					
					# Cannot turn capital state into colony
					NOT = {
						state_scope = {
							any_owned_province = {
								is_capital = yes
							}
						}
					}
				}
				
				owner = {
					country_event = {
						id = 92401
						days = 0
					}
				}
			}
			
			# Remove special flag core if invalid
			random_owned = {
				limit = {
					is_core = JAN
					
					OR = {
						is_colonial = yes
						
						state_scope = {
							any_owned_province = {
								is_capital = yes
							}
						}
					}
				}
				
				remove_core = JAN
			}
		}
	}
	
	# Set colony settlement policy
	option = {
		name = EVTOPTD92400
		
		add_core = JAN
		
		owner = {
			random_owned = {
				limit = {
					is_core = JAN
					is_colonial = yes
				}
				
				province_event = {
					id = 91520
					days = 0
				}
			}
		}
		
		remove_core = JAN
	}
	
	# Create Supply Depot
	option = {
		name = EVTOPTE92400
		
		add_core = JAN
		
		owner = {
			random_owned = {
				limit = {
					is_core = JAN
				}
				
				owner = {
					any_owned = {
						limit = {
							has_province_modifier = supply_depot
						}
						
						remove_province_modifier = supply_depot
					}
				}
				
				add_province_modifier = {
					name = supply_depot
					duration = -1
				}
			}
		}
		
		remove_core = JAN
	}
	
	# Change agricultural RGO
	option = {
		name = EVTOPTF92400
		
		add_core = JAN
		
		owner = {
			random_owned = {
				limit = {
					owner = { money = 250000 }
					
					is_core = JAN
					
					NOT = {
						OR = {
							trade_goods = coal
							trade_goods = iron
							trade_goods = sulphur
							trade_goods = oil
							trade_goods = rubber
							trade_goods = tropical_wood
							trade_goods = market_goods
							trade_goods = precious_metal
						}
					}
					
					NOT = { has_province_modifier = switched_production }
				}
				
				province_event = {
					id = 92403
					days = 0
				}
			}
		}
		
		remove_core = JAN
	}
	
	# Integrate State
	option = {
		name = EVTOPTG92400
		
		add_core = JAN
		
		owner = {
			random_owned = {
				limit = {
					is_core = JAN
					has_global_flag = province_selector_integration_enabled
					
					owner = {
						NOT = { has_modifier = active_integration }
					}
				}
				
				owner = {
					add_country_modifier = {
						name = active_integration
						duration = 3650
					}
				}
				
				state_scope = {
					any_owned = {
						add_province_modifier = {
							name = integrating_province
							duration = -1
						}
					}
				}
			}
		}
		
		remove_core = JAN
	}
}

# Event for player to confirm statehood removal
country_event = {
	id = 92401
	title = EVTNAME92401
	desc = EVTDESC92401
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA92401
	}
	
	option = {
		name = EVTOPTB92401
		
		random_owned = {
			limit = {
				is_core = JAN
				is_colonial = no
			}
			
			CIV = {
				country_event = {
					id = 92402
					days = 0
				}
			}
			
			state_scope = {
				any_owned = {
					secede_province = CIV
				}
			}
		}
	}
}

# State integration complete
province_event = {
	id = 92404
	title = EVTNAME92404
	desc = EVTDESC92404
	
	trigger = {
		owner = {
			NOT = { has_country_modifier = active_integration }
		}
		
		has_province_modifier = integrating_province
	}
	
	immediate = {
		state_scope = {
			any_owned = {
				limit = {
					has_province_modifier = integrating_province
				}
				
				remove_province_modifier = integrating_province
				
				add_core = THIS
			}
		}
	}
	
	option = {
		name = EVTOPTA92404
	}
}

# Event disrupting state integration if occupied
province_event = {
	id = 92405
	title = EVTNAME92405
	desc = EVTDESC92405
	
	trigger = {
		has_province_modifier = integrating_province
		NOT = { controlled_by = THIS }
	}
	
	immediate = {
		state_scope = {
			any_owned = {
				remove_province_modifier = integrating_province
			}
		}
		
		owner = {
			remove_country_modifier = active_integration
		}
	}
	
	option = {
		name = EVTOPTA92406
	}
}
